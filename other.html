def to_ordinal(number)

  last_two_digits = number % 100

  if last_two_digits == 11 || last_two_digits == 12 || last_two_digits == 13

    return "th"

  elsif number % 10 == 1

    return "st"

  elsif number % 10 == 2

    return "nd"

  elsif number % 10 == 3

    return "rd"

  else

    return "th"

  end

end
